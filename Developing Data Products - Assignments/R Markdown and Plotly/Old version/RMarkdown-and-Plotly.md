library(plotly)
library(ggplot2)
data(ToothGrowth)
plot <- ggplot(ToothGrowth, aes(as.factor(dose), len)) +
        geom_boxplot(aes(fill=as.factor(dose))) + 
        facet_grid(.~supp) +
        labs (x = "Dose Amount (mg/day)",
              y = "Length of odontoblasts",
              title = "Length vs. Dose by Supplement") +
        theme(legend.position = "none")
        
ggplotly(plot)
